基于shiny的天气查询系统

By 刘备备

Published 2017-04-17

Contents

思想很简单，基于Rvest函数制作爬虫，应用shiny搭建平台展示，源代码如下，测试本地可运行，但deploy到shinyapps出错，若你有好的思路，欢迎与我交流

library(shiny)
library(rvest)

sidebarPanel(
  textInput("city","city:","河南 商丘 城区"),

服务端：

library(shiny)
library(rvest)

shinyServer(function(input, output) {

datasetInput <- reactive({

  df1()
 })
  #接收按钮事件df1

observeEvent(input$gocity, {
  cat("Showing", input$city, "\n")
 })

df <- eventReactive(input$gocity, {
  xx=datasetInput()
  xx[,input$city]

 })
# 定义查询城市的按钮事件df
observeEvent(input$updata, {
  cat("Showing", input$new, "\n")
 })

df1 <- eventReactive(input$updata, {
 web<-read_html("http://www.weather.com.cn/textFC/henan.shtml",encoding="UTF-8")
 position<-web %>% html_nodes("div.conMidtab3")%>%html_nodes("a")%>%html_attr("href")
 position<-na.omit(position)
 position<-as.vector(position)
 position<-position[!duplicated(position)]

 tt<-list()
 for(i in position){
   web<-read_html(i,encoding = "UTF-8")
   pos1<-web%>%html_nodes("div.crumbs.fl")%>%html_nodes("a")%>%html_text()
   pos2<-web%>%html_nodes("div.crumbs.fl")%>%html_nodes("span")%>%html_text()
   pos2<-pos2[[3]]
   cityname<-paste(pos1[[1]],pos1[[2]],pos2,sep=" ")
   tianqi<-web%>%html_nodes("script")%>%html_text()
   tianqi<-tianqi[[4]]
   rules<-gregexpr("[0-9]+日[0-9]+时.{10,20}风",tianqi)
   tianqi<-regmatches(tianqi,rules)
   tianqi<-unlist(tianqi)
   tianqi<-gsub("[nd][0-9][0-9],","",tianqi)
   tt[[cityname]]<-tianqi
 }

 xx<-do.call(cbind,tt)

})
#定义按钮事件df1，爬虫，爬虫涉及rvest包，正则表达式，xml解析
#等内容，不在详细备注


output$tianqi <- renderTable({
 df()
})
#接收按钮事件df，输出到ui端
})
