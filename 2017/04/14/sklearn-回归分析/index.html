算法理解有回归理论和R方面的内容，再次不在详细描述

import numpy as np
import pandas as pd
from datetime import date, datetime
from pandasql import sqldf
from sklearn import linear_model
from sklearn import metrics
import matplotlib.pyplot as plt

print datetime.today()
linear = linear_model.LinearRegression()
#导入模型

df = pd.read_excel('C:/Users/Administrator/
Desktop/11.xls')#读取本地数据

# print df[["mpg"]]
# print df[["wt"]]
linear.fit(df[["wt", "cyl"]], df[["mpg"]])#建模
print "R",linear.score(df[["wt", "cyl"]], df[["mpg"]])
#输出R方

print('Coefficient: n', linear.coef_)
#输出回归系数
print('Intercept: n', linear.intercept_)
#输出截距项

ypred = linear.predict(df[["wt", "cyl"]])
#输出预测结果
print list(ypred)

print "MSE:", metrics.mean_squared_error
(df[["mpg"]], ypred)
#输出均方误差
print "RMSE:", 
np.sqrt(metrics.mean_squared_error(df[["mpg"]], ypred))
#输出均方误差平方根

注意：没有输出回归系数检验，F检验的函数，没有回归四项基本假设的检验函数，没有多重共线性的检验函数。需要自己查公式写吧
