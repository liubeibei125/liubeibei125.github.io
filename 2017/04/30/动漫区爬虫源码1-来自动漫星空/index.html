动漫区爬虫源码1-来自动漫星空

2017-04-30

采用动漫爬虫包rvest，创新点，写进文件以utf-8，写入特定的文件头内容，能够直接上传hexo

源码：

url<-"http://acg.gamersky.com/pic/wallpaper/pc/"

web<-read_html(url,encoding="UTF-8")
t<-web%>%html_nodes("li.img")%>%html_nodes("a")%>%
html_attr("href")



for(j in 1:length(t)){

web22<-read_html(t[j],encoding = "UTF-8")
text<-web22%>%html_node("title")%>%html_text()
text<-gsub(" ","",text)
tt<-paste("---\n","title: ",text,"\n","date: 
2017-04-28\n","categories: 动漫\n","---\n",sep="")
tt<-enc2utf8(tt)
f<-file(paste("H:/
dfgh/",text,".md",sep=""),"a",encoding = "UTF-8")
cat(tt, file=f)
##写进头文件，编码utf-8

z<-list()
tryCatch({
for(i in 1:6){
 if(i==1){
web2<-read_html(t[j],encoding = "UTF-8")
z[[i]]<-web2%>%html_nodes("img.picact")%>%html_attr("src")
 }else{
url2<-gsub(".shtml","",t[j])
web2<-read_html(paste
(url2,"_",i,".shtml",sep=""),encoding = "UTF-8")
z[[i]]<-web2%>%html_nodes("img.picact")%>%html_attr("src")

}}
},error=function(e){
 cat("error:",conditionMessage(e),"\n")
})
z<-unlist(z)
z<-paste("![alt](",z,")",sep="")
write.table(z,f,row.names = F,col.names = F,quote = F)
close(f)
}
