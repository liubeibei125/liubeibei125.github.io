动漫区爬虫源码2-来自58壁纸

2017-04-30

采用动漫爬虫包rvest，创新点，写进文件以utf-8，写入特定的文件头内容，能够直接上传hexo，读取gb2312编码的网站

源码：

library(rvest)
zz<-list()
for(i in 1:25){
url<-paste("http://www.bz55.com/dongmanbizhi/
list_3_",i,".html",sep="")
web<-read_html(url,encoding="GBK")
zz[[i]]<-web%>%html_nodes("div.list_txt")%>%
html_nodes("ul")%>%html_nodes("li")%>%html_nodes("a")
%>%html_attr("href")
zz[[i]]<-zz[[i]][1:12]
}

zz<-unlist(zz)

for(i in 1:length(zz)){
 z<-list()
 url<-paste("http://www.bz55.com",zz[i],sep="")
 web<-read_html(url,encoding = "GBK")
 text<-web%>%html_nodes("p")%>%html_nodes("img")%>%
 html_attr("alt")
 text<-text[1]
 text<-gsub(" ","",text)
 tt<-paste("---\n","title: ",text,"\n","date: 
 2017-04-29\n","categories: 动漫\n","---\n",sep="")
 tt<-enc2utf8(tt)
 f<-file(paste("H:/
 dfgh/",text,".md",sep=""),"a",encoding = "UTF-8")
 cat(tt, file=f)

 tryCatch({
 url<-paste("http://www.bz55.com",zz[i],sep="")
 web<-read_html(url,encoding = "GBK")
 z[[i]]<-web%>%html_nodes("p")%>%html_nodes("img")%>%
 html_attr("src")
 },error=function(e){
cat("error:",conditionMessage(e),"\n")
})
z<-unlist(z)
z<-paste("![alt](","http://www.bz55.com",z,")",sep="")
write.table(z,f,row.names = F,col.names = F,quote = F)
close(f)

}
